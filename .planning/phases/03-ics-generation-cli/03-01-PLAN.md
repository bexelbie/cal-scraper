```python
from icalendar import Calendar, Event as IcsEvent

cal = Calendar()
cal.add('prodid', '...')
cal.add('version', '2.0')

vevent = IcsEvent()
vevent.add('summary', 'Title')
vevent.add('dtstart', datetime_or_date)  # date obj → DATE, datetime obj → DATE-TIME
vevent.add('dtend', datetime_or_date)
vevent.add('location', 'Place')
vevent.add('description', 'Text')
vevent.add('url', 'https://...')
vevent.add('uid', 'unique@domain')
vevent.add('dtstamp', datetime.now(tz=...))

cal.add_component(vevent)
cal.add_missing_timezones()  # Adds VTIMEZONE for referenced timezones
ics_bytes = cal.to_ical()  # Returns bytes
```
